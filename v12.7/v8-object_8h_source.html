858Local<Data> Object::GetInternalField(int index) {
859#ifndef V8_ENABLE_CHECKS
860 using A = internal::Address;
861 using I = internal::Internals;
862 A obj = internal::ValueHelper::ValueAsAddress(this);
863 // Fast path: If the object is a plain JSObject, which is the common case, we
864 // know where to find the internal fields and can return the value directly.
865 int instance_type = I::GetInstanceType(obj);
866 if (I::CanHaveInternalField(instance_type)) {
867 int offset = I::kJSAPIObjectWithEmbedderSlotsHeaderSize +
868 (I::kEmbedderDataSlotSize * index);
869 A value = I::ReadRawField<A>(obj, offset);
870#ifdef V8_COMPRESS_POINTERS
871 // We read the full pointer value and then decompress it in order to avoid
872 // dealing with potential endiannes issues.
873 value = I::DecompressTaggedField(obj, static_cast<uint32_t>(value));
874#endif
875
876 auto isolate = reinterpret_cast<v8::Isolate*>(
877 internal::IsolateFromNeverReadOnlySpaceObject(obj));
878 return Local<Data>::New(isolate, value);
879 }
880#endif
881 return SlowGetInternalField(index);
882}

884void* Object::GetAlignedPointerFromInternalField(v8::Isolate* isolate,
885 int index) {
886#if !defined(V8_ENABLE_CHECKS)
887 using A = internal::Address;
888 using I = internal::Internals;
889 A obj = internal::ValueHelper::ValueAsAddress(this);
890 // Fast path: If the object is a plain JSObject, which is the common case, we
891 // know where to find the internal fields and can return the value directly.
892 auto instance_type = I::GetInstanceType(obj);
893 if (V8_LIKELY(I::CanHaveInternalField(instance_type))) {
894 int offset = I::kJSAPIObjectWithEmbedderSlotsHeaderSize +
895 (I::kEmbedderDataSlotSize * index) +
896 I::kEmbedderDataSlotExternalPointerOffset;
897 A value =
898 I::ReadExternalPointerField<internal::kEmbedderDataSlotPayloadTag>(
899 isolate, obj, offset);
900 return reinterpret_cast<void*>(value);
901 }
902#endif
903 return SlowGetAlignedPointerFromInternalField(isolate, index);
904}
