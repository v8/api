172 // Options determines how the trace buffer stores data.
173 enum TraceRecordMode {
174  // Record until the trace buffer is full.
175  RECORD_UNTIL_FULL,
176 
177  // Record until the user ends the trace. The trace buffer is a fixed size
178  // and we use it as a ring buffer during recording.
179  RECORD_CONTINUOUSLY,
180 
181  // Record until the trace buffer is full, but with a huge buffer size.
182  RECORD_AS_MUCH_AS_POSSIBLE,
183 
184  // Echo to console. Events are discarded.
185  ECHO_TO_CONSOLE,
186 };
