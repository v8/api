149// Options determines how the trace buffer stores data.
150enum TraceRecordMode {
151 // Record until the trace buffer is full.
152 RECORD_UNTIL_FULL,
153
154 // Record until the user ends the trace. The trace buffer is a fixed size
155 // and we use it as a ring buffer during recording.
156 RECORD_CONTINUOUSLY,
157
158 // Record until the trace buffer is full, but with a huge buffer size.
159 RECORD_AS_MUCH_AS_POSSIBLE,
160
161 // Echo to console. Events are discarded.
162 ECHO_TO_CONSOLE,
163};
