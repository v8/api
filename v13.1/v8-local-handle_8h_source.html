478template <typename T>
479class LocalVector {
480 private:
481 using element_type = internal::LocalUnchecked<T>;
482
483#ifdef V8_ENABLE_DIRECT_HANDLE
484 using allocator_type = internal::StrongRootAllocator<element_type>;
485
486 static allocator_type make_allocator(Isolate* isolate) noexcept {
487 return allocator_type(isolate);
488 }
489#else
490 using allocator_type = std::allocator<element_type>;
491
492 static allocator_type make_allocator(Isolate* isolate) noexcept {
493 return allocator_type();
494 }
495#endif // V8_ENABLE_DIRECT_HANDLE
496
497 using vector_type = std::vector<element_type, allocator_type>;
498
499 public:
500 using value_type = Local<T>;
501 using reference = value_type&;
502 using const_reference = const value_type&;
503 using size_type = size_t;
504 using difference_type = ptrdiff_t;
505 using iterator =
506 internal::WrappedIterator<typename vector_type::iterator, Local<T>>;
507 using const_iterator =
508 internal::WrappedIterator<typename vector_type::const_iterator,
509 const Local<T>>;
510
511 explicit LocalVector(Isolate* isolate) : backing_(make_allocator(isolate)) {}
512 LocalVector(Isolate* isolate, size_t n)
513 : backing_(n, make_allocator(isolate)) {}
514 explicit LocalVector(Isolate* isolate, std::initializer_list<Local<T>> init)
515 : backing_(make_allocator(isolate)) {
516 if (init.size() == 0) return;
517 backing_.reserve(init.size());
518 backing_.insert(backing_.end(), init.begin(), init.end());
519 }
520
521 iterator begin() noexcept { return iterator(backing_.begin()); }
522 const_iterator begin() const noexcept {
523 return const_iterator(backing_.begin());
524 }
525 iterator end() noexcept { return iterator(backing_.end()); }
526 const_iterator end() const noexcept { return const_iterator(backing_.end()); }
527
528 size_t size() const noexcept { return backing_.size(); }
529 bool empty() const noexcept { return backing_.empty(); }
530 void reserve(size_t n) { backing_.reserve(n); }
531 void shrink_to_fit() { backing_.shrink_to_fit(); }
532
533 Local<T>& operator[](size_t n) { return backing_[n]; }
534 const Local<T>& operator[](size_t n) const { return backing_[n]; }
535
536 Local<T>& at(size_t n) { return backing_.at(n); }
537 const Local<T>& at(size_t n) const { return backing_.at(n); }
538
539 Local<T>& front() { return backing_.front(); }
540 const Local<T>& front() const { return backing_.front(); }
541 Local<T>& back() { return backing_.back(); }
542 const Local<T>& back() const { return backing_.back(); }
543
544 Local<T>* data() noexcept { return backing_.data(); }
545 const Local<T>* data() const noexcept { return backing_.data(); }
546
547 iterator insert(const_iterator pos, const Local<T>& value) {
548 return iterator(backing_.insert(pos.base(), value));
549 }
550
551 template <typename InputIt>
552 iterator insert(const_iterator pos, InputIt first, InputIt last) {
553 return iterator(backing_.insert(pos.base(), first, last));
554 }
555
556 iterator insert(const_iterator pos, std::initializer_list<Local<T>> init) {
557 return iterator(backing_.insert(pos.base(), init.begin(), init.end()));
558 }
559
560 LocalVector<T>& operator=(std::initializer_list<Local<T>> init) {
561 backing_.clear();
562 backing_.reserve(init.size());
563 backing_.insert(backing_.end(), init.begin(), init.end());
564 return *this;
565 }
566
567 void push_back(const Local<T>& x) { backing_.push_back(x); }
568 void pop_back() { backing_.pop_back(); }
569 void emplace_back(const Local<T>& x) { backing_.emplace_back(x); }
570
571 void clear() noexcept { backing_.clear(); }
572 void resize(size_t n) { backing_.resize(n); }
573 void swap(LocalVector<T>& other) { backing_.swap(other.backing_); }
574
575 friend bool operator==(const LocalVector<T>& x, const LocalVector<T>& y) {
576 return x.backing_ == y.backing_;
577 }
578 friend bool operator!=(const LocalVector<T>& x, const LocalVector<T>& y) {
579 return x.backing_ != y.backing_;
580 }
581 friend bool operator<(const LocalVector<T>& x, const LocalVector<T>& y) {
582 return x.backing_ < y.backing_;
583 }
584 friend bool operator>(const LocalVector<T>& x, const LocalVector<T>& y) {
585 return x.backing_ > y.backing_;
586 }
587 friend bool operator<=(const LocalVector<T>& x, const LocalVector<T>& y) {
588 return x.backing_ <= y.backing_;
589 }
590 friend bool operator>=(const LocalVector<T>& x, const LocalVector<T>& y) {
591 return x.backing_ >= y.backing_;
592 }
593
594 private:
595 vector_type backing_;
596};
