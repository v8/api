58class CompressedPointer final {
59 public:
60 using Storage = uint32_t;
61
62 V8_INLINE CompressedPointer() : value_(0u) {}
63 V8_INLINE explicit CompressedPointer(const void* ptr)
64 : value_(Compress(ptr)) {}
65 V8_INLINE explicit CompressedPointer(std::nullptr_t) : value_(0u) {}
66 V8_INLINE explicit CompressedPointer(SentinelPointer)
67 : value_(kCompressedSentinel) {}
68
69 V8_INLINE const void* Load() const { return Decompress(value_); }
70 V8_INLINE const void* LoadAtomic() const {
71 return Decompress(
72 reinterpret_cast<const std::atomic<Storage>&>(value_).load(
73 std::memory_order_relaxed));
74 }
75
76 V8_INLINE void Store(const void* ptr) { value_ = Compress(ptr); }
77 V8_INLINE void StoreAtomic(const void* value) {
78 reinterpret_cast<std::atomic<Storage>&>(value_).store(
79 Compress(value), std::memory_order_relaxed);
80 }
81
82 V8_INLINE void Clear() { value_ = 0u; }
83 V8_INLINE bool IsCleared() const { return !value_; }
84
85 V8_INLINE friend bool operator==(CompressedPointer a, CompressedPointer b) {
86 return a.value_ == b.value_;
87 }
88
89 static V8_INLINE Storage Compress(const void* ptr) {
90 static_assert(
91 SentinelPointer::kSentinelValue == 0b10,
92 "The compression scheme relies on the sentinel encoded as 0b10");
93 static constexpr size_t kGigaCageMask =
94 ~(api_constants::kCagedHeapReservationAlignment - 1);
95
96 CPPGC_DCHECK(CageBaseGlobal::IsSet());
97 const uintptr_t base = CageBaseGlobal::Get();
98 CPPGC_DCHECK(!ptr || ptr == kSentinelPointer ||
99 (base & kGigaCageMask) ==
100 (reinterpret_cast<uintptr_t>(ptr) & kGigaCageMask));
101
102 const auto uptr = reinterpret_cast<uintptr_t>(ptr);
103 // Truncate the pointer and shift right by one.
104 auto compressed = static_cast<Storage>(uptr) >> 1;
105 // If the pointer is regular, set the most significant bit.
106 if (V8_LIKELY(compressed > 1)) {
107 CPPGC_DCHECK((reinterpret_cast<uintptr_t>(ptr) &
108 (api_constants::kAllocationGranularity - 1)) == 0);
109 compressed |= 0x80000000;
110 }
111 return compressed;
112 }
113
114 static V8_INLINE void* Decompress(Storage ptr) {
115 CPPGC_DCHECK(CageBaseGlobal::IsSet());
116 const uintptr_t base = CageBaseGlobal::Get();
117 // Sign extend the pointer and shift left by one.
118 const int64_t mask = static_cast<int64_t>(static_cast<int32_t>(ptr)) << 1;
119 return reinterpret_cast<void*>(mask & base);
120 }
121
122 private:
123 static constexpr Storage kCompressedSentinel =
124 SentinelPointer::kSentinelValue >> 1;
125 // All constructors initialize `value_`. Do not add a default value here as it
126 // results in a non-atomic write on some builds, even when the atomic version
127 // of the constructor is used.
128 Storage value_;
129};
