360 struct V8_EXPORT CachedData {
361 enum BufferPolicy { BufferNotOwned, BufferOwned };
362
363 CachedData()
364 : data(nullptr),
365 length(0),
366 rejected(false),
367 buffer_policy(BufferNotOwned) {}
368
369 // If buffer_policy is BufferNotOwned, the caller keeps the ownership of
370 // data and guarantees that it stays alive until the CachedData object is
371 // destroyed. If the policy is BufferOwned, the given data will be deleted
372 // (with delete[]) when the CachedData object is destroyed.
373 CachedData(const uint8_t* data, int length,
374 BufferPolicy buffer_policy = BufferNotOwned);
375 ~CachedData();
376 // TODO(marja): Async compilation; add constructors which take a callback
377 // which will be called when V8 no longer needs the data.
378 const uint8_t* data;
379 int length;
380 bool rejected;
381 BufferPolicy buffer_policy;
382
383 // Prevent copying.
384 CachedData(const CachedData&) = delete;
385 CachedData& operator=(const CachedData&) = delete;
386 };
