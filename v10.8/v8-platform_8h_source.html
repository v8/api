527// Opaque type representing a handle to a shared memory region.
528using PlatformSharedMemoryHandle = intptr_t;
529static constexpr PlatformSharedMemoryHandle kInvalidSharedMemoryHandle = -1;
530
531// Conversion routines from the platform-dependent shared memory identifiers
532// into the opaque PlatformSharedMemoryHandle type. These use the underlying
533// types (e.g. unsigned int) instead of the typedef'd ones (e.g. mach_port_t)
534// to avoid pulling in large OS header files into this header file. Instead,
535// the users of these routines are expected to include the respecitve OS
536// headers in addition to this one.
537#if V8_OS_MACOS
538// Convert between a shared memory handle and a mach_port_t referencing a memory
539// entry object.
540inline PlatformSharedMemoryHandle SharedMemoryHandleFromMachMemoryEntry(
541 unsigned int port) {
542 return static_cast<PlatformSharedMemoryHandle>(port);
543}
544inline unsigned int MachMemoryEntryFromSharedMemoryHandle(
545 PlatformSharedMemoryHandle handle) {
546 return static_cast<unsigned int>(handle);
547}
548#elif V8_OS_FUCHSIA
549// Convert between a shared memory handle and a zx_handle_t to a VMO.
550inline PlatformSharedMemoryHandle SharedMemoryHandleFromVMO(uint32_t handle) {
551 return static_cast<PlatformSharedMemoryHandle>(handle);
552}
553inline uint32_t VMOFromSharedMemoryHandle(PlatformSharedMemoryHandle handle) {
554 return static_cast<uint32_t>(handle);
555}
556#elif V8_OS_WIN
557// Convert between a shared memory handle and a Windows HANDLE to a file mapping
558// object.
559inline PlatformSharedMemoryHandle SharedMemoryHandleFromFileMapping(
560 void* handle) {
561 return reinterpret_cast<PlatformSharedMemoryHandle>(handle);
562}
563inline void* FileMappingFromSharedMemoryHandle(
564 PlatformSharedMemoryHandle handle) {
565 return reinterpret_cast<void*>(handle);
566}
567#else
568// Convert between a shared memory handle and a file descriptor.
569inline PlatformSharedMemoryHandle SharedMemoryHandleFromFileDescriptor(int fd) {
570 return static_cast<PlatformSharedMemoryHandle>(fd);
571}
572inline int FileDescriptorFromSharedMemoryHandle(
573 PlatformSharedMemoryHandle handle) {
574 return static_cast<int>(handle);
575}
576#endif
