73// Smi constants for systems where tagged pointer is a 32-bit value.
74template <>
75struct SmiTagging<4> {
76 enum { kSmiShiftSize = 0, kSmiValueSize = 31 };
77
78 static constexpr intptr_t kSmiMinValue =
79 static_cast<intptr_t>(kUintptrAllBitsSet << (kSmiValueSize - 1));
80 static constexpr intptr_t kSmiMaxValue = -(kSmiMinValue + 1);
81
82 V8_INLINE static int SmiToInt(const internal::Address value) {
83 int shift_bits = kSmiTagSize + kSmiShiftSize;
84 // Truncate and shift down (requires >> to be sign extending).
85 return static_cast<int32_t>(static_cast<uint32_t>(value)) >> shift_bits;
86 }
87 V8_INLINE static constexpr bool IsValidSmi(intptr_t value) {
88 // Is value in range [kSmiMinValue, kSmiMaxValue].
89 // Use unsigned operations in order to avoid undefined behaviour in case of
90 // signed integer overflow.
91 return (static_cast<uintptr_t>(value) -
92 static_cast<uintptr_t>(kSmiMinValue)) <=
93 (static_cast<uintptr_t>(kSmiMaxValue) -
94 static_cast<uintptr_t>(kSmiMinValue));
95 }
96};

98// Smi constants for systems where tagged pointer is a 64-bit value.

100struct SmiTagging<8> {
101 enum { kSmiShiftSize = 31, kSmiValueSize = 32 };

103 static constexpr intptr_t kSmiMinValue =
104 static_cast<intptr_t>(kUintptrAllBitsSet << (kSmiValueSize - 1));

107 V8_INLINE static int SmiToInt(const internal::Address value) {
108 int shift_bits = kSmiTagSize + kSmiShiftSize;

190#ifdef V8_OS_ANDROID
191// On Android, most 64-bit devices seem to be configured with only 39 bits of
192// virtual address space for userspace. As such, limit the sandbox to 128GB (a
193// quarter of the total available address space).
194constexpr size_t kSandboxSizeLog2 = 37; // 128 GB
195#else
196// Everywhere else use a 1TB sandbox.
197constexpr size_t kSandboxSizeLog2 = 40; // 1 TB
198#endif // V8_OS_ANDROID
199constexpr size_t kSandboxSize = 1ULL << kSandboxSizeLog2;
200
201// Required alignment of the sandbox. For simplicity, we require the
202// size of the guard regions to be a multiple of this, so that this specifies
203// the alignment of the sandbox including and excluding surrounding guard
204// regions. The alignment requirement is due to the pointer compression cage
205// being located at the start of the sandbox.
206constexpr size_t kSandboxAlignment = kPtrComprCageBaseAlignment;
207
208// Sandboxed pointers are stored inside the heap as offset from the sandbox
209// base shifted to the left. This way, it is guaranteed that the offset is
210// smaller than the sandbox size after shifting it to the right again. This
211// constant specifies the shift amount.
212constexpr uint64_t kSandboxedPointerShift = 64 - kSandboxSizeLog2;
213
214// Size of the guard regions surrounding the sandbox. This assumes a worst-case
215// scenario of a 32-bit unsigned index used to access an array of 64-bit
216// values.
217constexpr size_t kSandboxGuardRegionSize = 32ULL * GB;
218
219static_assert((kSandboxGuardRegionSize % kSandboxAlignment) == 0,
220 "The size of the guard regions around the sandbox must be a "
221 "multiple of its required alignment.");
222
223// Minimum size of the sandbox, excluding the guard regions surrounding it. If
224// the virtual memory reservation for the sandbox fails, its size is currently
225// halved until either the reservation succeeds or the minimum size is reached.
226// A minimum of 32GB allows the 4GB pointer compression region as well as the
227// ArrayBuffer partition and two 10GB Wasm memory cages to fit into the
228// sandbox. 32GB should also be the minimum possible size of the userspace
229// address space as there are some machine configurations with only 36 virtual
230// address bits.
231constexpr size_t kSandboxMinimumSize = 32ULL * GB;
232
233static_assert(kSandboxMinimumSize <= kSandboxSize,
234 "The minimal size of the sandbox must be smaller or equal to the "
235 "regular size.");
236
237// On OSes where reserving virtual memory is too expensive to reserve the
238// entire address space backing the sandbox, notably Windows pre 8.1, we create
239// a partially reserved sandbox that doesn't actually reserve most of the
240// memory, and so doesn't have the desired security properties as unrelated
241// memory allocations could end up inside of it, but which still ensures that
242// objects that should be located inside the sandbox are allocated within
243// kSandboxSize bytes from the start of the sandbox. The minimum size of the
244// region that is actually reserved for such a sandbox is specified by this
245// constant and should be big enough to contain the pointer compression cage as
246// well as the ArrayBuffer partition.
247constexpr size_t kSandboxMinimumReservationSize = 8ULL * GB;
248
249static_assert(kSandboxMinimumSize > kPtrComprCageReservationSize,
250 "The sandbox must be larger than the pointer compression cage "
251 "contained within it.");
252static_assert(kSandboxMinimumReservationSize > kPtrComprCageReservationSize,
253 "The minimum reservation size for a sandbox must be larger than "
254 "the pointer compression cage contained within it.");
255
256// For now, even if the sandbox is enabled, we still allow backing stores to be
257// allocated outside of it as fallback. This will simplify the initial rollout.
258// However, if sandboxed pointers are also enabled, we must always place
259// backing stores inside the sandbox as they will be referenced though them.

266// The size of the virtual memory reservation for an external pointer table.
267// This determines the maximum number of entries in a table. Using a maximum
268// size allows omitting bounds checks on table accesses if the indices are
269// guaranteed (e.g. through shifting) to be below the maximum index. This
270// value must be a power of two.
271static const size_t kExternalPointerTableReservationSize = 128 * MB;

273// The maximum number of entries in an external pointer table.

288// If sandboxed external pointers are enabled, these tag values will be ORed
289// with the external pointers in the external pointer table to prevent use of
290// pointers of the wrong type. When a pointer is loaded, it is ANDed with the
291// inverse of the expected type's tag. The tags are constructed in a way that
292// guarantees that a failed type check will result in one or more of the top
293// bits of the pointer to be set, rendering the pointer inacessible. Besides
294// the type tag bits (48 through 62), the tags also have the GC mark bit (63)
295// set, so that the mark bit is automatically set when a pointer is written
296// into the external pointer table (in which case it is clearly alive) and is
297// cleared when the pointer is loaded. The exception to this is the free entry
298// tag, which doesn't have the mark bit set, as the entry is not alive. This
299// construction allows performing the type check and removing GC marking bits
300// (the MSB) from the pointer at the same time.
301// Note: this scheme assumes a 48-bit address space and will likely break if
302// more virtual address bits are used.
303constexpr uint64_t kExternalPointerTagMask = 0xffff000000000000;
304constexpr uint64_t kExternalPointerTagShift = 48;
305#define MAKE_TAG(v) (static_cast<uint64_t>(v) << kExternalPointerTagShift)
306// clang-format off
307enum ExternalPointerTag : uint64_t {
308 kExternalPointerNullTag = MAKE_TAG(0b0000000000000000),
309 kExternalPointerFreeEntryTag = MAKE_TAG(0b0111111110000000),
310 kExternalStringResourceTag = MAKE_TAG(0b1000000011111111),
311 kExternalStringResourceDataTag = MAKE_TAG(0b1000000101111111),
312 kForeignForeignAddressTag = MAKE_TAG(0b1000000110111111),
313 kNativeContextMicrotaskQueueTag = MAKE_TAG(0b1000000111011111),
314 kEmbedderDataSlotPayloadTag = MAKE_TAG(0b1000000111101111),
315 kCodeEntryPointTag = MAKE_TAG(0b1000000111110111),
316 kExternalObjectValueTag = MAKE_TAG(0b1000000111111011),
317};
318// clang-format on
319#undef MAKE_TAG

541 template <typename T>
542 V8_INLINE static T ReadRawField(internal::Address heap_object_ptr,
543 int offset) {
544 internal::Address addr = heap_object_ptr + offset - kHeapObjectTag;
545#ifdef V8_COMPRESS_POINTERS
546 if (sizeof(T) > kApiTaggedSize) {
547 // TODO(ishell, v8:8875): When pointer compression is enabled 8-byte size
548 // fields (external pointers, doubles and BigInt data) are only
549 // kTaggedSize aligned so we have to use unaligned pointer friendly way of
550 // accessing them in order to avoid undefined behavior in C++ code.
551 T r;
552 memcpy(&r, reinterpret_cast<void*>(addr), sizeof(T));
553 return r;
554 }
555#endif
556 return *reinterpret_cast<const T*>(addr);
557 }
