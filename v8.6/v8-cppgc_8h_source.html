60JSMemberBase& JSMemberBase::CopyImpl(const JSMemberBase& other) {
61 if (this != &other) {
62 Reset();
63 if (!other.IsEmpty()) {
64 Copy(&other.val_, &val_);
65 }
66 }
67 return *this;
68}

70JSMemberBase& JSMemberBase::MoveImpl(JSMemberBase&& other) {
71 if (this != &other) {
72 // No call to Reset() as Move() will conditionally reset itself when needed,
73 // and otherwise reuse the internal meta data.
74 Move(&other.val_, &val_);
75 }
76 return *this;
77}
