37 // See api_constants for an explanation of the constants.
38 std::atomic<uint16_t>* atomic_mutable_bitfield =
39 reinterpret_cast<std::atomic<uint16_t>*>(

41 reinterpret_cast<const uint8_t*>(payload) -

166template <typename T>
167class MakeGarbageCollectedTrait : public MakeGarbageCollectedTraitBase<T> {
168 public:
169 template <typename... Args>
170 static T* Call(AllocationHandle& handle, Args&&... args) {
171 void* memory =
172 MakeGarbageCollectedTraitBase<T>::Allocate(handle, sizeof(T));
173 T* object = ::new (memory) T(std::forward<Args>(args)...);
174 MakeGarbageCollectedTraitBase<T>::MarkObjectAsFullyConstructed(object);
175 return object;
176 }
