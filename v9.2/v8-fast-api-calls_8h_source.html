199class CTypeInfo {
200 public:
201 enum class Type : uint8_t {
202 kVoid,
203 kBool,
204 kInt32,
205 kUint32,
206 kInt64,
207 kUint64,
208 kFloat32,
209 kFloat64,
210 kV8Value,
211 kApiObject, // This will be deprecated once all users have
212 // migrated from v8::ApiObject to v8::Local<v8::Value>.
213 };
214
215 // kCallbackOptionsType is not part of the Type enum
216 // because it is only used internally. Use value 255 that is larger
217 // than any valid Type enum.
218 static constexpr Type kCallbackOptionsType = Type(255);
219

221 kNone = 0,
222 };
223

225 : type_(type), flags_(flags) {}
226
227 constexpr Type GetType() const { return type_; }
228
229 constexpr Flags GetFlags() const { return flags_; }
230
231 private:
232 Type type_;
233 Flags flags_;
234};

236class V8_EXPORT CFunctionInfo {
237 public:
238 // Construct a struct to hold a CFunction's type information.
239 // |return_info| describes the function's return type.
240 // |arg_info| is an array of |arg_count| CTypeInfos describing the
241 // arguments. Only the last argument may be of the special type
242 // CTypeInfo::kCallbackOptionsType.
243 CFunctionInfo(const CTypeInfo& return_info, unsigned int arg_count,
244 const CTypeInfo* arg_info);
245
246 const CTypeInfo& ReturnInfo() const { return return_info_; }
247
248 // The argument count, not including the v8::FastApiCallbackOptions
249 // if present.
250 unsigned int ArgumentCount() const {
251 return HasOptions() ? arg_count_ - 1 : arg_count_;
252 }
253
254 // |index| must be less than ArgumentCount().
255 // Note: if the last argument passed on construction of CFunctionInfo
256 // has type CTypeInfo::kCallbackOptionsType, it is not included in
257 // ArgumentCount().
258 const CTypeInfo& ArgumentInfo(unsigned int index) const;
259
260 bool HasOptions() const {
261 // The options arg is always the last one.
262 return arg_count_ > 0 && arg_info_[arg_count_ - 1].GetType() ==
263 CTypeInfo::kCallbackOptionsType;
264 }
265
266 private:
267 const CTypeInfo return_info_;
268 const unsigned int arg_count_;
269 const CTypeInfo* arg_info_;
270};

370template <typename RetBuilder, typename... ArgBuilders>
371class CFunctionInfoImpl : public CFunctionInfo {
372 static constexpr int kOptionsArgCount =
373 count<FastApiCallbackOptions&, ArgBuilders...>();
374 static constexpr int kReceiverCount = 1;
375
376 static_assert(kOptionsArgCount == 0 || kOptionsArgCount == 1,
377 "Only one options parameter is supported.");
378

380 "The receiver or the options argument is missing.");

447template <typename T, CTypeInfo::Flags... Flags>
448class CTypeInfoBuilder {
449 public:
450 using BaseType = T;
451
452 static constexpr CTypeInfo Build() {
453 // Get the flags and merge in any additional flags.
454 uint8_t flags = uint8_t(TypeInfoHelper<T>::Flags());
455 int unused[] = {0, (flags |= uint8_t(Flags), 0)...};
456 // With C++17, we could use a "..." fold expression over a parameter pack.
457 // Since we're still using C++14, we have to evaluate an OR expresion while
458 // constructing an unused list of 0's. This applies the binary operator
459 // for each value in Flags.
460 (void)unused;
461
462 // Return the same type with the merged flags.
463 return CTypeInfo(TypeInfoHelper<T>::Type(), CTypeInfo::Flags(flags));
464 }
465};

467template <typename RetBuilder, typename... ArgBuilders>
468class CFunctionBuilderWithFunction {
469 public:
470 explicit constexpr CFunctionBuilderWithFunction(const void* fn) : fn_(fn) {}
471
472 template <CTypeInfo::Flags... Flags>
473 constexpr auto Ret() {
474 return CFunctionBuilderWithFunction<
475 CTypeInfoBuilder<typename RetBuilder::BaseType, Flags...>,
476 ArgBuilders...>(fn_);
477 }
478
479 template <unsigned int N, CTypeInfo::Flags... Flags>
480 constexpr auto Arg() {
481 // Return a copy of the builder with the Nth arg builder merged with
482 // template parameter pack Flags.
483 return ArgImpl<N, Flags...>(
484 std::make_index_sequence<sizeof...(ArgBuilders)>());
485 }
486
487 auto Build() {
488 static CFunctionInfoImpl<RetBuilder, ArgBuilders...> instance;
489 return CFunction(fn_, &instance);
490 }
491
492 private:
493 template <bool Merge, unsigned int N, CTypeInfo::Flags... Flags>
494 struct GetArgBuilder;
495
496 // Returns the same ArgBuilder as the one at index N, including its flags.
497 // Flags in the template parameter pack are ignored.
498 template <unsigned int N, CTypeInfo::Flags... Flags>
499 struct GetArgBuilder<false, N, Flags...> {
500 using type =
501 typename std::tuple_element<N, std::tuple<ArgBuilders...>>::type;
502 };
503
504 // Returns an ArgBuilder with the same base type as the one at index N,
505 // but merges the flags with the flags in the template parameter pack.
506 template <unsigned int N, CTypeInfo::Flags... Flags>
507 struct GetArgBuilder<true, N, Flags...> {
508 using type = CTypeInfoBuilder<
509 typename std::tuple_element<N,
510 std::tuple<ArgBuilders...>>::type::BaseType,
511 std::tuple_element<N, std::tuple<ArgBuilders...>>::type::Build()
512 .GetFlags(),
513 Flags...>;
514 };
515
516 // Return a copy of the CFunctionBuilder, but merges the Flags on ArgBuilder
517 // index N with the new Flags passed in the template parameter pack.
518 template <unsigned int N, CTypeInfo::Flags... Flags, size_t... I>
519 constexpr auto ArgImpl(std::index_sequence<I...>) {
520 return CFunctionBuilderWithFunction<
521 RetBuilder, typename GetArgBuilder<N == I, I, Flags...>::type...>(fn_);
522 }
523
524 const void* fn_;
525};
