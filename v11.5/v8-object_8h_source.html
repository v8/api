735Local<Value> Object::GetInternalField(int index) {
736#ifndef V8_ENABLE_CHECKS
737 using A = internal::Address;
738 using I = internal::Internals;
739 A obj = internal::ValueHelper::ValueAsAddress(this);
740 // Fast path: If the object is a plain JSObject, which is the common case, we
741 // know where to find the internal fields and can return the value directly.
742 int instance_type = I::GetInstanceType(obj);
743 if (I::CanHaveInternalField(instance_type)) {
744 int offset = I::kJSObjectHeaderSize + (I::kEmbedderDataSlotSize * index);
745 A value = I::ReadRawField<A>(obj, offset);
746#ifdef V8_COMPRESS_POINTERS
747 // We read the full pointer value and then decompress it in order to avoid
748 // dealing with potential endiannes issues.
749 value = I::DecompressTaggedField(obj, static_cast<uint32_t>(value));
750#endif
751
752 auto isolate = reinterpret_cast<v8::Isolate*>(
753 internal::IsolateFromNeverReadOnlySpaceObject(obj));
754 return Local<Value>::New(isolate, value);
755 }
756#endif
757 return SlowGetInternalField(index);
758}
759
760void* Object::GetAlignedPointerFromInternalField(int index) {
761#if !defined(V8_ENABLE_CHECKS)
762 using A = internal::Address;
763 using I = internal::Internals;
764 A obj = internal::ValueHelper::ValueAsAddress(this);
765 // Fast path: If the object is a plain JSObject, which is the common case, we
766 // know where to find the internal fields and can return the value directly.
767 auto instance_type = I::GetInstanceType(obj);
768 if (I::CanHaveInternalField(instance_type)) {
769 int offset = I::kJSObjectHeaderSize + (I::kEmbedderDataSlotSize * index) +
770 I::kEmbedderDataSlotExternalPointerOffset;
771 Isolate* isolate = I::GetIsolateForSandbox(obj);
772 A value =
773 I::ReadExternalPointerField<internal::kEmbedderDataSlotPayloadTag>(
774 isolate, obj, offset);
775 return reinterpret_cast<void*>(value);
776 }
777#endif
778 return SlowGetAlignedPointerFromInternalField(index);
779}
