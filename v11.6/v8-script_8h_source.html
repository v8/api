376 struct V8_EXPORT CachedData {
377 enum BufferPolicy { BufferNotOwned, BufferOwned };
378
379 CachedData()
380 : data(nullptr),
381 length(0),
382 rejected(false),
383 buffer_policy(BufferNotOwned) {}
384
385 // If buffer_policy is BufferNotOwned, the caller keeps the ownership of
386 // data and guarantees that it stays alive until the CachedData object is
387 // destroyed. If the policy is BufferOwned, the given data will be deleted
388 // (with delete[]) when the CachedData object is destroyed.
389 CachedData(const uint8_t* data, int length,
390 BufferPolicy buffer_policy = BufferNotOwned);
391 ~CachedData();
392 // TODO(marja): Async compilation; add constructors which take a callback
393 // which will be called when V8 no longer needs the data.
394 const uint8_t* data;
395 int length;
396 bool rejected;
397 BufferPolicy buffer_policy;
398
399 // Prevent copying.
400 CachedData(const CachedData&) = delete;
401 CachedData& operator=(const CachedData&) = delete;
402 };
