182 template <typename HeapHandleCallback>
183 static V8_INLINE WriteBarrierType
184 GetWriteBarrierType(const TracedReferenceBase& ref,
185 WriteBarrierParams& params, HeapHandleCallback callback) {
186 if (ref.IsEmpty()) return WriteBarrierType::kNone;
187
188 if (V8_LIKELY(!cppgc::internal::WriteBarrier::
189 IsAnyIncrementalOrConcurrentMarking())) {
190 return cppgc::internal::WriteBarrier::Type::kNone;
191 }
192 cppgc::HeapHandle& handle = callback();
193 if (!cppgc::subtle::HeapState::IsMarking(handle)) {
194 return cppgc::internal::WriteBarrier::Type::kNone;
195 }
196 params.heap = &handle;
197#if V8_ENABLE_CHECKS
198 params.type = cppgc::internal::WriteBarrier::Type::kMarking;
199#endif // !V8_ENABLE_CHECKS
200 return cppgc::internal::WriteBarrier::Type::kMarking;
201 }
