413Local<Value> Context::GetEmbedderData(int index) {
414#ifndef V8_ENABLE_CHECKS
415 using A = internal::Address;
416 using I = internal::Internals;
417 A ctx = internal::ValueHelper::ValueAsAddress(this);
418 A embedder_data =
419 I::ReadTaggedPointerField(ctx, I::kNativeContextEmbedderDataOffset);
420 int value_offset =
421 I::kEmbedderDataArrayHeaderSize + (I::kEmbedderDataSlotSize * index);
422 A value = I::ReadRawField<A>(embedder_data, value_offset);
423#ifdef V8_COMPRESS_POINTERS
424 // We read the full pointer value and then decompress it in order to avoid
425 // dealing with potential endiannes issues.
426 value = I::DecompressTaggedField(embedder_data, static_cast<uint32_t>(value));
427#endif
428
429 auto isolate = reinterpret_cast<v8::Isolate*>(
430 internal::IsolateFromNeverReadOnlySpaceObject(ctx));
431 return Local<Value>::New(isolate, value);
432#else
433 return SlowGetEmbedderData(index);
434#endif
435}

437void* Context::GetAlignedPointerFromEmbedderData(int index) {
438#if !defined(V8_ENABLE_CHECKS)
439 using A = internal::Address;
440 using I = internal::Internals;
441 A ctx = internal::ValueHelper::ValueAsAddress(this);
442 A embedder_data =
443 I::ReadTaggedPointerField(ctx, I::kNativeContextEmbedderDataOffset);
444 int value_offset = I::kEmbedderDataArrayHeaderSize +
445 (I::kEmbedderDataSlotSize * index) +
446 I::kEmbedderDataSlotExternalPointerOffset;
447 Isolate* isolate = I::GetIsolateForSandbox(ctx);
448 return reinterpret_cast<void*>(
449 I::ReadExternalPointerField<internal::kEmbedderDataSlotPayloadTag>(
450 isolate, embedder_data, value_offset));
451#else
452 return SlowGetAlignedPointerFromEmbedderData(index);
453#endif
454}

456template <class T>
457MaybeLocal<T> Context::GetDataFromSnapshotOnce(size_t index) {
458 auto slot = GetDataFromSnapshotOnce(index);
459 if (slot) {
460 internal::PerformCastCheck(
461 internal::ValueHelper::SlotAsValue<T, false>(slot));
462 }
463 return Local<T>::FromSlot(slot);
464}
