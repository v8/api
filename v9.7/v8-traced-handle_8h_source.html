597void TracedReferenceBase::SetWrapperClassId(uint16_t class_id) {
598 using I = internal::Internals;
599 if (IsEmpty()) return;
600 internal::Address* obj = reinterpret_cast<internal::Address*>(val_);
601 uint8_t* addr = reinterpret_cast<uint8_t*>(obj) + I::kNodeClassIdOffset;
602 *reinterpret_cast<uint16_t*>(addr) = class_id;
603}
604
605uint16_t TracedReferenceBase::WrapperClassId() const {
606 using I = internal::Internals;
607 if (IsEmpty()) return 0;
608 internal::Address* obj = reinterpret_cast<internal::Address*>(val_);
609 uint8_t* addr = reinterpret_cast<uint8_t*>(obj) + I::kNodeClassIdOffset;
610 return *reinterpret_cast<uint16_t*>(addr);
611}
