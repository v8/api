66// Smi constants for systems where tagged pointer is a 32-bit value.
67template <>
68struct SmiTagging<4> {
69 enum { kSmiShiftSize = 0, kSmiValueSize = 31 };
70
71 static constexpr intptr_t kSmiMinValue =
72 static_cast<intptr_t>(kUintptrAllBitsSet << (kSmiValueSize - 1));
73 static constexpr intptr_t kSmiMaxValue = -(kSmiMinValue + 1);
74
75 V8_INLINE static int SmiToInt(const internal::Address value) {
76 int shift_bits = kSmiTagSize + kSmiShiftSize;
77 // Truncate and shift down (requires >> to be sign extending).
78 return static_cast<int32_t>(static_cast<uint32_t>(value)) >> shift_bits;
79 }
80 V8_INLINE static constexpr bool IsValidSmi(intptr_t value) {
81 // Is value in range [kSmiMinValue, kSmiMaxValue].
82 // Use unsigned operations in order to avoid undefined behaviour in case of
83 // signed integer overflow.
84 return (static_cast<uintptr_t>(value) -
85 static_cast<uintptr_t>(kSmiMinValue)) <=
86 (static_cast<uintptr_t>(kSmiMaxValue) -
87 static_cast<uintptr_t>(kSmiMinValue));
88 }
89};

134// If the heap sandbox is enabled, these tag values will be ORed with the
135// external pointers in the external pointer table to prevent use of pointers of
136// the wrong type. When a pointer is loaded, it is ANDed with the inverse of the
137// expected type's tag. The tags are constructed in a way that guarantees that a
138// failed type check will result in one or more of the top bits of the pointer
139// to be set, rendering the pointer inacessible. This construction allows
140// performing the type check and removing GC marking bits from the pointer at
141// the same time.
142enum ExternalPointerTag : uint64_t {
143 kExternalPointerNullTag = 0x0000000000000000,
144 kExternalStringResourceTag = 0x00ff000000000000, // 0b000000011111111
145 kExternalStringResourceDataTag = 0x017f000000000000, // 0b000000101111111
146 kForeignForeignAddressTag = 0x01bf000000000000, // 0b000000110111111
147 kNativeContextMicrotaskQueueTag = 0x01df000000000000, // 0b000000111011111
148 kEmbedderDataSlotPayloadTag = 0x01ef000000000000, // 0b000000111101111
149 kCodeEntryPointTag = 0x01f7000000000000, // 0b000000111110111
150};

395 template <typename T>
396 V8_INLINE static T ReadRawField(internal::Address heap_object_ptr,
397 int offset) {
398 internal::Address addr = heap_object_ptr + offset - kHeapObjectTag;
399#ifdef V8_COMPRESS_POINTERS
400 if (sizeof(T) > kApiTaggedSize) {
401 // TODO(ishell, v8:8875): When pointer compression is enabled 8-byte size
402 // fields (external pointers, doubles and BigInt data) are only
403 // kTaggedSize aligned so we have to use unaligned pointer friendly way of
404 // accessing them in order to avoid undefined behavior in C++ code.
405 T r;
406 memcpy(&r, reinterpret_cast<void*>(addr), sizeof(T));
407 return r;
408 }
409#endif
410 return *reinterpret_cast<const T*>(addr);
411 }

499#define GB (1ULL << 30)
500#define TB (1ULL << 40)
501
502// Size of the virtual memory cage, excluding the guard regions surrounding it.
503constexpr size_t kVirtualMemoryCageSizeLog2 = 40; // 1 TB
504constexpr size_t kVirtualMemoryCageSize = 1ULL << kVirtualMemoryCageSizeLog2;
505
506// Required alignment of the virtual memory cage. For simplicity, we require the
507// size of the guard regions to be a multiple of this, so that this specifies
508// the alignment of the cage including and excluding surrounding guard regions.
509// The alignment requirement is due to the pointer compression cage being
510// located at the start of the virtual memory cage.
511constexpr size_t kVirtualMemoryCageAlignment =
512 Internals::kPtrComprCageBaseAlignment;
513
514#ifdef V8_CAGED_POINTERS
515// CagedPointers are guaranteed to point into the virtual memory cage. This is
516// achieved by storing them as offset from the cage base rather than as raw
517// pointers.
518using CagedPointer_t = Address;
519
520// For efficiency, the offset is stored shifted to the left, so that
521// it is guaranteed that the offset is smaller than the cage size after
522// shifting it to the right again. This constant specifies the shift amount.
523constexpr uint64_t kCagedPointerShift = 64 - kVirtualMemoryCageSizeLog2;
524#endif
525
526// Size of the guard regions surrounding the virtual memory cage. This assumes a
527// worst-case scenario of a 32-bit unsigned index being used to access an array
528// of 64-bit values.
529constexpr size_t kVirtualMemoryCageGuardRegionSize = 32ULL * GB;
530
531static_assert((kVirtualMemoryCageGuardRegionSize %
532 kVirtualMemoryCageAlignment) == 0,
533 "The size of the virtual memory cage guard region must be a "
534 "multiple of its required alignment.");
535
536// Minimum size of the virtual memory cage, excluding the guard regions
537// surrounding it. If the cage reservation fails, its size is currently halved
538// until either the reservation succeeds or the minimum size is reached. A
539// minimum of 32GB allows the 4GB pointer compression region as well as the
540// ArrayBuffer partition and two 10GB WASM memory cages to fit into the cage.
541// 32GB should also be the minimum possible size of the userspace address space
542// as there are some machine configurations with only 36 virtual address bits.
543constexpr size_t kVirtualMemoryCageMinimumSize = 32ULL * GB;
544
545static_assert(kVirtualMemoryCageMinimumSize <= kVirtualMemoryCageSize,
546 "The minimal size of the virtual memory cage must be smaller or "
547 "equal to the regular size.");
548
549// On OSes where reservation virtual memory is too expensive to create a real
550// cage, notably Windows pre 8.1, we create a fake cage that doesn't actually
551// reserve most of the memory, and so doesn't have the desired security
552// properties, but still ensures that objects that should be located inside the
553// cage are allocated within kVirtualMemoryCageSize bytes from the start of the
554// cage, and so appear to be inside the cage. The minimum size of the virtual
555// memory range that is actually reserved for a fake cage is specified by this
556// constant and should be big enough to contain the pointer compression region
557// as well as the ArrayBuffer partition.
558constexpr size_t kFakeVirtualMemoryCageMinReservationSize = 8ULL * GB;
559
560static_assert(kVirtualMemoryCageMinimumSize >
561 Internals::kPtrComprCageReservationSize,
562 "The virtual memory cage must be larger than the pointer "
563 "compression cage contained within it.");
564static_assert(kFakeVirtualMemoryCageMinReservationSize >
565 Internals::kPtrComprCageReservationSize,
566 "The reservation for a fake virtual memory cage must be larger "
567 "than the pointer compression cage contained within it.");
568
