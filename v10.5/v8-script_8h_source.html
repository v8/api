362 struct V8_EXPORT CachedData {
363 enum BufferPolicy { BufferNotOwned, BufferOwned };
364
365 CachedData()
366 : data(nullptr),
367 length(0),
368 rejected(false),
369 buffer_policy(BufferNotOwned) {}
370
371 // If buffer_policy is BufferNotOwned, the caller keeps the ownership of
372 // data and guarantees that it stays alive until the CachedData object is
373 // destroyed. If the policy is BufferOwned, the given data will be deleted
374 // (with delete[]) when the CachedData object is destroyed.
375 CachedData(const uint8_t* data, int length,
376 BufferPolicy buffer_policy = BufferNotOwned);
377 ~CachedData();
378 // TODO(marja): Async compilation; add constructors which take a callback
379 // which will be called when V8 no longer needs the data.
380 const uint8_t* data;
381 int length;
382 bool rejected;
383 BufferPolicy buffer_policy;
384
385 // Prevent copying.
386 CachedData(const CachedData&) = delete;
387 CachedData& operator=(const CachedData&) = delete;
388 };
