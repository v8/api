485class LocalVector {
486 private:
487 using element_type = internal::LocalUnchecked<T>;
488
489#ifdef V8_ENABLE_DIRECT_LOCAL
490 using allocator_type = internal::StrongRootAllocator<element_type>;
491
492 static allocator_type make_allocator(Isolate* isolate) noexcept {
493 return allocator_type(isolate);
494 }
495#else
496 using allocator_type = std::allocator<element_type>;
497
498 static allocator_type make_allocator(Isolate* isolate) noexcept {
499 return allocator_type();
500 }
501#endif // V8_ENABLE_DIRECT_LOCAL
502
503 using vector_type = std::vector<element_type, allocator_type>;
504
505 public:
506 using value_type = Local<T>;
507 using reference = value_type&;
508 using const_reference = const value_type&;
509 using size_type = size_t;
510 using difference_type = ptrdiff_t;
511 using iterator =
512 internal::WrappedIterator<typename vector_type::iterator, Local<T>>;
513 using const_iterator =
514 internal::WrappedIterator<typename vector_type::const_iterator,
515 const Local<T>>;
516
517 explicit LocalVector(Isolate* isolate) : backing_(make_allocator(isolate)) {}
518 LocalVector(Isolate* isolate, size_t n)
519 : backing_(n, make_allocator(isolate)) {}
520 explicit LocalVector(Isolate* isolate, std::initializer_list<Local<T>> init)
521 : backing_(make_allocator(isolate)) {
522 if (init.size() == 0) return;
523 backing_.reserve(init.size());
524 backing_.insert(backing_.end(), init.begin(), init.end());
525 }
526
527 iterator begin() noexcept { return iterator(backing_.begin()); }
528 const_iterator begin() const noexcept {
529 return const_iterator(backing_.begin());
530 }

532 const_iterator end() const noexcept { return const_iterator(backing_.end()); }
533
534 size_t size() const noexcept { return backing_.size(); }
535 bool empty() const noexcept { return backing_.empty(); }
536 void reserve(size_t n) { backing_.reserve(n); }
537 void shrink_to_fit() { backing_.shrink_to_fit(); }
538
539 Local<T>& operator[](size_t n) { return backing_[n]; }
540 const Local<T>& operator[](size_t n) const { return backing_[n]; }
541
542 Local<T>& at(size_t n) { return backing_.at(n); }
543 const Local<T>& at(size_t n) const { return backing_.at(n); }
544
545 Local<T>& front() { return backing_.front(); }
546 const Local<T>& front() const { return backing_.front(); }
547 Local<T>& back() { return backing_.back(); }
548 const Local<T>& back() const { return backing_.back(); }
549
550 Local<T>* data() noexcept { return backing_.data(); }
551 const Local<T>* data() const noexcept { return backing_.data(); }
552
553 iterator insert(const_iterator pos, const Local<T>& value) {
554 return iterator(backing_.insert(pos.base(), value));
555 }
556
557 template <typename InputIt>
558 iterator insert(const_iterator pos, InputIt first, InputIt last) {
559 return iterator(backing_.insert(pos.base(), first, last));
560 }

564 }
565

567 backing_.clear();
568 backing_.insert(backing_.end(), init.begin(), init.end());

570 }
571
572 void push_back(const Local<T>& x) { backing_.push_back(x); }
573 void pop_back() { backing_.pop_back(); }
574 void emplace_back(const Local<T>& x) { backing_.emplace_back(x); }
575
576 void clear() noexcept { backing_.clear(); }
577 void resize(size_t n) { backing_.resize(n); }
578 void swap(LocalVector<T>& other) { backing_.swap(other.backing_); }
579
580 friend bool operator==(const LocalVector<T>& x, const LocalVector<T>& y) {
581 return x.backing_ == y.backing_;
582 }
583 friend bool operator!=(const LocalVector<T>& x, const LocalVector<T>& y) {
584 return x.backing_ != y.backing_;
585 }
586 friend bool operator<(const LocalVector<T>& x, const LocalVector<T>& y) {
587 return x.backing_ < y.backing_;
588 }
589 friend bool operator>(const LocalVector<T>& x, const LocalVector<T>& y) {
590 return x.backing_ > y.backing_;
591 }
592 friend bool operator<=(const LocalVector<T>& x, const LocalVector<T>& y) {
593 return x.backing_ <= y.backing_;
594 }
595 friend bool operator>=(const LocalVector<T>& x, const LocalVector<T>& y) {
596 return x.backing_ >= y.backing_;
597 }
598
599 private:
600 vector_type backing_;
601};
