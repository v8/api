894String::ExternalStringResource* String::GetExternalStringResource() const {
895 using A = internal::Address;
896 using I = internal::Internals;
897 A obj = internal::ValueHelper::ValueAsAddress(this);
898
899 ExternalStringResource* result;
900 if (I::IsExternalTwoByteString(I::GetInstanceType(obj))) {
901 Isolate* isolate = I::GetIsolateForSandbox(obj);
902 A value = I::ReadExternalPointerField<internal::kExternalStringResourceTag>(
903 isolate, obj, I::kStringResourceOffset);
904 result = reinterpret_cast<String::ExternalStringResource*>(value);
905 } else {
906 result = GetExternalStringResourceSlow();
907 }
908#ifdef V8_ENABLE_CHECKS
909 VerifyExternalStringResource(result);
910#endif
911 return result;
912}

914String::ExternalStringResourceBase* String::GetExternalStringResourceBase(
915 v8::Isolate* isolate, String::Encoding* encoding_out) const {
916 using A = internal::Address;
917 using I = internal::Internals;
918 A obj = internal::ValueHelper::ValueAsAddress(this);
919 int type = I::GetInstanceType(obj) & I::kStringRepresentationAndEncodingMask;
920 *encoding_out = static_cast<Encoding>(type & I::kStringEncodingMask);
921 ExternalStringResourceBase* resource;
922 if (type == I::kExternalOneByteRepresentationTag ||
923 type == I::kExternalTwoByteRepresentationTag) {
924 A value = I::ReadExternalPointerField<internal::kExternalStringResourceTag>(
925 isolate, obj, I::kStringResourceOffset);
926 resource = reinterpret_cast<ExternalStringResourceBase*>(value);
927 } else {
928 resource = GetExternalStringResourceBaseSlow(encoding_out);
929 }
930#ifdef V8_ENABLE_CHECKS
931 VerifyExternalStringResourceBase(resource, *encoding_out);
932#endif
933 return resource;
934}
