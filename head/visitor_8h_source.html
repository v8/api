38  static_assert(sizeof(T), "Pointee type must be fully defined.");
39  static_assert(internal::IsGarbageCollectedType<T>::value,
40  "T must be GarabgeCollected or GarbageCollectedMixin type");
41 
42  const T* value = weak_member.GetRawAtomic();
43 
44  // Bailout assumes that WeakMember emits write barrier.
45  if (!value) {
46  return;
47  }
48 
49  // TODO(chromium:1056170): DCHECK (or similar) for deleted values as they
50  // should come in at a different path.

52  &HandleWeak<WeakMember<T>>, &weak_member);

61  static_assert(internal::IsGarbageCollectedType<PointeeType>::value,

89  // processed immediately, and (b) only processed conservatively if not

112  template <typename T, void (T::*method)(const LivenessBroker&)>
113  static void WeakCallbackMethodDelegate(const LivenessBroker& info,
114  const void* self) {
115  // Callback is registered through a potential const Trace method but needs
116  // to be able to modify fields. See HandleWeak.
117  (const_cast<T*>(static_cast<const T*>(self))->*method)(info);
118  }

120  template <typename PointerType>
121  static void HandleWeak(const LivenessBroker& info, const void* object) {
122  const PointerType* weak = static_cast<const PointerType*>(object);
123  const auto* raw = weak->Get();
124  if (raw && !info.IsHeapObjectAlive(raw)) {
125  // Object is passed down through the marker as const. Alternatives are
126  // - non-const Trace method;
127  // - mutable pointer in MemberBase;
128  const_cast<PointerType*>(weak)->Clear();
129  }
130  }
