537// Opaque type representing a handle to a shared memory region.
538using PlatformSharedMemoryHandle = intptr_t;
539static constexpr PlatformSharedMemoryHandle kInvalidSharedMemoryHandle = -1;
540
541// Conversion routines from the platform-dependent shared memory identifiers
542// into the opaque PlatformSharedMemoryHandle type. These use the underlying
543// types (e.g. unsigned int) instead of the typedef'd ones (e.g. mach_port_t)
544// to avoid pulling in large OS header files into this header file. Instead,
545// the users of these routines are expected to include the respecitve OS
546// headers in addition to this one.
547#if V8_OS_DARWIN
548// Convert between a shared memory handle and a mach_port_t referencing a memory
549// entry object.
550inline PlatformSharedMemoryHandle SharedMemoryHandleFromMachMemoryEntry(
551 unsigned int port) {
552 return static_cast<PlatformSharedMemoryHandle>(port);
553}
554inline unsigned int MachMemoryEntryFromSharedMemoryHandle(
555 PlatformSharedMemoryHandle handle) {
556 return static_cast<unsigned int>(handle);
557}
558#elif V8_OS_FUCHSIA
559// Convert between a shared memory handle and a zx_handle_t to a VMO.
560inline PlatformSharedMemoryHandle SharedMemoryHandleFromVMO(uint32_t handle) {
561 return static_cast<PlatformSharedMemoryHandle>(handle);
562}
563inline uint32_t VMOFromSharedMemoryHandle(PlatformSharedMemoryHandle handle) {
564 return static_cast<uint32_t>(handle);
565}
566#elif V8_OS_WIN
567// Convert between a shared memory handle and a Windows HANDLE to a file mapping
568// object.
569inline PlatformSharedMemoryHandle SharedMemoryHandleFromFileMapping(
570 void* handle) {
571 return reinterpret_cast<PlatformSharedMemoryHandle>(handle);
572}
573inline void* FileMappingFromSharedMemoryHandle(
574 PlatformSharedMemoryHandle handle) {
575 return reinterpret_cast<void*>(handle);
576}
577#else
578// Convert between a shared memory handle and a file descriptor.
579inline PlatformSharedMemoryHandle SharedMemoryHandleFromFileDescriptor(int fd) {
580 return static_cast<PlatformSharedMemoryHandle>(fd);
581}
582inline int FileDescriptorFromSharedMemoryHandle(
583 PlatformSharedMemoryHandle handle) {
584 return static_cast<int>(handle);
585}
586#endif
