172 class CTypeInfo {
173  public:
174  enum class Type : char {
175  kVoid,
176  kBool,
177  kInt32,
178  kUint32,
179  kInt64,
180  kUint64,
181  kFloat32,
182  kFloat64,
183  kUnwrappedApiObject,
184  };
185 
186  enum ArgFlags : char {
187  None = 0,
188  IsArrayBit = 1 << 0, // This argument is first in an array of values.
189  };
190 
191  static CTypeInfo FromWrapperType(const void* wrapper_type_info,
192  ArgFlags flags = ArgFlags::None) {
193  uintptr_t wrapper_type_info_ptr =
194  reinterpret_cast<uintptr_t>(wrapper_type_info);
195  // Check that the lower kIsWrapperTypeBit bits are 0's.
196  CHECK_EQ(
197  wrapper_type_info_ptr & ~(static_cast<uintptr_t>(~0)
198  << static_cast<uintptr_t>(kIsWrapperTypeBit)),
199  0);
200  // TODO(mslekova): Refactor the manual bit manipulations to use
201  // PointerWithPayload instead.
202  return CTypeInfo(wrapper_type_info_ptr | flags | kIsWrapperTypeBit);
203  }
204 
205  static constexpr CTypeInfo FromCType(Type ctype,
206  ArgFlags flags = ArgFlags::None) {
207  // ctype cannot be Type::kUnwrappedApiObject.
208  return CTypeInfo(
209  ((static_cast<uintptr_t>(ctype) << kTypeOffset) & kTypeMask) | flags);
210  }
211 
212  const void* GetWrapperInfo() const;
213 
214  constexpr Type GetType() const {
215  if (payload_ & kIsWrapperTypeBit) {
216  return Type::kUnwrappedApiObject;
217  }
218  return static_cast<Type>((payload_ & kTypeMask) >> kTypeOffset);
219  }
220 
221  constexpr bool IsArray() const { return payload_ & ArgFlags::IsArrayBit; }
222 
223  private:
224  explicit constexpr CTypeInfo(uintptr_t payload) : payload_(payload) {}
225 
226  // That must be the last bit after ArgFlags.
227  static constexpr uintptr_t kIsWrapperTypeBit = 1 << 1;
228  static constexpr uintptr_t kWrapperTypeInfoMask = static_cast<uintptr_t>(~0)
229  << 2;
230 
231  static constexpr unsigned int kTypeOffset = kIsWrapperTypeBit;
232  static constexpr unsigned int kTypeSize = 8 - kTypeOffset;
233  static constexpr uintptr_t kTypeMask =
234  (~(static_cast<uintptr_t>(~0) << kTypeSize)) << kTypeOffset;
235 
236  const uintptr_t payload_;
237 };
