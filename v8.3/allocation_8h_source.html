25 static inline void MarkObjectAsFullyConstructed(const void* payload) {
26 // See api_constants for an explanation of the constants.
27 std::atomic<uint16_t>* atomic_mutable_bitfield =
28 reinterpret_cast<std::atomic<uint16_t>*>(
29 const_cast<uint16_t*>(reinterpret_cast<const uint16_t*>(
30 reinterpret_cast<const uint8_t*>(payload) -
31 api_constants::kFullyConstructedBitFieldOffsetFromPayload)));
32 uint16_t value = atomic_mutable_bitfield->load(std::memory_order_relaxed);

34 atomic_mutable_bitfield->store(value, std::memory_order_release);

44// Users with custom allocation needs (e.g. overriding size) should override
45// MakeGarbageCollectedTrait (see below) and inherit their trait from
46// MakeGarbageCollectedTraitBase to get access to low-level primitives.
47template <typename T>
48class MakeGarbageCollectedTraitBase
49 : private internal::MakeGarbageCollectedTraitInternal {
50 protected:
51 // Allocates an object of |size| bytes on |heap|.
52 //
53 // TODO(mlippautz): Allow specifying arena for specific embedder uses.
54 static void* Allocate(Heap* heap, size_t size) {
55 return internal::MakeGarbageCollectedTraitInternal::Allocate(
56 heap, size, internal::GCInfoTrait<T>::Index());
57 }

60 // by the garbage collector.

68template <typename T>
69class MakeGarbageCollectedTrait : public MakeGarbageCollectedTraitBase<T> {
70 public:
71 template <typename... Args>
72 static T* Call(Heap* heap, Args&&... args) {
73 static_assert(internal::IsGarbageCollectedType<T>::value,
74 "T needs to be a garbage collected object");
75 void* memory = MakeGarbageCollectedTraitBase<T>::Allocate(heap, sizeof(T));
76 T* object = ::new (memory) T(std::forward<Args>(args)...);
77 MakeGarbageCollectedTraitBase<T>::MarkObjectAsFullyConstructed(object);
78 return object;
79 }
80};

82// Default MakeGarbageCollected: Constructs an instance of T, which is a garbage
83// collected type.
84template <typename T, typename... Args>
85T* MakeGarbageCollected(Heap* heap, Args&&... args) {
86 return MakeGarbageCollectedTrait<T>::Call(heap, std::forward<Args>(args)...);
