51 template <typename T>
52 void Trace(const Member<T>& member) {
53 const T* value = member.GetRawAtomic();
54 CPPGC_DCHECK(value != kSentinelPointer);
55 Trace(value);
56 }

63 template <typename T>
64 void Trace(const WeakMember<T>& weak_member) {
65 static_assert(sizeof(T), "Pointee type must be fully defined.");
66 static_assert(internal::IsGarbageCollectedType<T>::value,
67 "T must be GarbageCollected or GarbageCollectedMixin type");
68
69 const T* value = weak_member.GetRawAtomic();
70
71 // Bailout assumes that WeakMember emits write barrier.
72 if (!value) {
73 return;
74 }
75
76 // TODO(chromium:1056170): DCHECK (or similar) for deleted values as they
77 // should come in at a different path.
78 VisitWeak(value, TraceTrait<T>::GetTraceDescriptor(value),
79 &HandleWeak<WeakMember<T>>, &weak_member);
80 }

88 template <typename T>
89 void Trace(const T& object) {
90#if V8_ENABLE_CHECKS
91 // This object is embedded in potentially multiple nested objects. The
92 // outermost object must not be in construction as such objects are (a) not
93 // processed immediately, and (b) only processed conservatively if not
94 // otherwise possible.
95 CheckObjectNotInConstruction(&object);
96#endif // V8_ENABLE_CHECKS
97 TraceTrait<T>::Trace(this, &object);
98 }

128 template <typename T, void (T::*method)(const LivenessBroker&)>
129 static void WeakCallbackMethodDelegate(const LivenessBroker& info,
130 const void* self) {
131 // Callback is registered through a potential const Trace method but needs
132 // to be able to modify fields. See HandleWeak.
133 (const_cast<T*>(static_cast<const T*>(self))->*method)(info);
134 }

136 template <typename PointerType>
137 static void HandleWeak(const LivenessBroker& info, const void* object) {
138 const PointerType* weak = static_cast<const PointerType*>(object);
139 // Sentinel values are preserved for weak pointers.
140 if (*weak == kSentinelPointer) return;
141 const auto* raw = weak->Get();
142 if (!info.IsHeapObjectAlive(raw)) {
143 weak->ClearFromGC();
144 }
145 }

149 template <typename Persistent,
150 std::enable_if_t<Persistent::IsStrongPersistent::value>* = nullptr>
151 void TraceRoot(const Persistent& p, const SourceLocation& loc) {
152 using PointeeType = typename Persistent::PointeeType;
153 static_assert(sizeof(PointeeType),
154 "Persistent's pointee type must be fully defined");
155 static_assert(internal::IsGarbageCollectedType<PointeeType>::value,
156 "Persistent's pointee type must be GarbageCollected or "
157 "GarbageCollectedMixin");
158 if (!p.Get()) {
159 return;
160 }
161 VisitRoot(p.Get(), TraceTrait<PointeeType>::GetTraceDescriptor(p.Get()));
162 }
